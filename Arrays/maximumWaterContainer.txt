var maxArea = function(height) {
   let maxValue=0;
   for(let i=0;i<height.length;i++){
       for(let j=i+1;j<height.length;j++){
           let heights=Math.min(height[j],height[i]);
           let width=j-i;
           let area=heights*width;
           maxValue=Math.max(maxValue,area);
       }
   }
   return maxValue; 
};
